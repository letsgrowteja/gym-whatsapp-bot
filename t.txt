const express = require("express");
const cors = require("cors");
const axios = require("axios");
const cron = require("node-cron");
require("dotenv").config();

const { createClient } = require("@supabase/supabase-js");

// 🔗 Supabase
const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_KEY
);

const app = express();
app.use(cors());
app.use(express.json());

app.get("/", (req, res) => {
  res.send("Gym WhatsApp Bot Running 🚀");
});

// 🛑 OVERLAP LOCK
let isRunning = false;

// 🔁 RETRY FUNCTION
async function sendWithRetry(fn, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (err) {
      console.log(`⚠️ Retry ${i + 1}`);

      if (err.response?.status === 429) {
        await new Promise(res => setTimeout(res, 5000));
      } else {
        await new Promise(res => setTimeout(res, 2000));
      }
    }
  }
  console.log("❌ Failed after retries");
}

// 🔥 FORMAT MESSAGE (IMPORTANT)
function formatMessage(type, name, date) {
  if (type === "before_expiry") {
    return `Hi ${name}! *You've built the habit, now don't let it slip. *⚡Your membership set to renew on ${date}. Let’s keep that momentum going—renew now to keep your progress! 📈`;
  }
  if (type === "today_expiry") {
    return `Hi ${name}! Your membership ends today (${date}). *Remember—Consistency is what separates the best from the rest.*🔥 You are so *Close* to your *Goal*—don't let it slip now! *Renew today* to maintain your '*Active*' status and keep the *Grind going*. 🏋️‍♂️`;
  }
  if (type === "after_expiry") {
    return `Hi ${name}, *Don't let all that hard work go to waste* 📉. Your membership wrapped up on ${date}, *But your Goals are still waiting for You*. The best time to *Restart* is right *Now*. Ready to get back on *Track*? 👊`;
  }
  return "";
}

// 🔥 TEMPLATE SEND
async function sendTemplate(phone, name, expiry_date, templateName) {
  return sendWithRetry(async () => {
    await axios.post(
      `https://graph.facebook.com/v18.0/${process.env.PHONE_NUMBER_ID}/messages`,
      {
        messaging_product: "whatsapp",
        to: phone,
        type: "template",
        template: {
          name: templateName,
          language: { code: "en" },
          components: [
            {
              type: "body",
              parameters: [
                { type: "text", text: name },
                { type: "text", text: expiry_date }
              ]
            }
          ]
        }
      },
      {
        headers: {
          Authorization: `Bearer ${process.env.ACCESS_TOKEN}`,
          "Content-Type": "application/json",
        },
      }
    );

    console.log(`✅ ${templateName} sent to ${phone}`);
  });
}

// WRAPPERS
const sendBeforeExpiry = (p, n, d) => sendTemplate(p, n, d, "before_expiry");
const sendTodayExpiry  = (p, n, d) => sendTemplate(p, n, d, "today_expiry");
const sendAfterExpiry  = (p, n, d) => sendTemplate(p, n, d, "after_expiry");

// ➕ ADD MEMBER
app.post("/add-member", async (req, res) => {
  const { name, phone, expiry_date, gym_id } = req.body;

  if (!name || !phone || !expiry_date || !gym_id) {
    return res.status(400).json({ error: "All fields required" });
  }

  const { data, error } = await supabase.from("members").insert([
    { name, phone, expiry_date, gym_id, status: "pending" }
  ]);

  if (error) return res.status(500).json({ error });

  res.json({ message: "Member added", data });
});

// 🔁 CRON JOB
cron.schedule("*/1 * * * *", async () => {
  if (isRunning) return;

  isRunning = true;
  console.log("⏰ Running expiry check...");

  try {
    const today = new Date();

    const { data } = await supabase.from("members").select("*");

    if (!data) {
    console.log("❌ No members found");
    return;
    }

    for (let member of data) {
      const expiry = new Date(member.expiry_date);
      const diffDays = Math.ceil((expiry - today) / (1000 * 60 * 60 * 24));

      console.log(`👤 ${member.name} → ${diffDays}`);

      try {
        let type = null;

        if (diffDays === 3 && member.status !== "before_sent") {
          type = "before_expiry";
          await sendBeforeExpiry(member.phone, member.name, member.expiry_date);
          await supabase.from("members").update({ status: "before_sent" }).eq("id", member.id);
        }
        else if (diffDays === 0 && member.status !== "today_sent") {
          type = "today_expiry";
          await sendTodayExpiry(member.phone, member.name, member.expiry_date);
          await supabase.from("members").update({ status: "today_sent" }).eq("id", member.id);
        }
        else if (diffDays < 0 && member.status !== "after_sent") {
          type = "after_expiry";
          await sendAfterExpiry(member.phone, member.name, member.expiry_date);
          await supabase.from("members").update({ status: "after_sent" }).eq("id", member.id);
        }

        // 💾 SAVE SAME MESSAGE TO DB
        if (type) {
          const msg = formatMessage(type, member.name, member.expiry_date);

          await supabase.from("messages").insert([
            {
              phone: member.phone,
              message: msg,
              sender: "business",
              type: "text",
              gym_id: member.gym_id,
            }
          ]);

          console.log("💾 Saved to DB");
        }

      } catch (err) {
        console.log("❌ Member failed:", member.name);
      }

      await new Promise(res => setTimeout(res, 1000));
    }

  } catch (err) {
    console.log("❌ CRON ERROR:", err);
  }

  isRunning = false;
});

// 🔔 WEBHOOK VERIFY
app.get("/webhook", (req, res) => {
  const VERIFY_TOKEN = "gymbot265";

  if (
    req.query["hub.mode"] &&
    req.query["hub.verify_token"] === VERIFY_TOKEN
  ) {
    return res.status(200).send(req.query["hub.challenge"]);
  }

  res.sendStatus(403);
});

// 📩 RECEIVE MESSAGE
app.post("/webhook", async (req, res) => {
  try {
    const messageData = req.body.entry?.[0]?.changes?.[0]?.value?.messages?.[0];

    if (!messageData) return res.sendStatus(200);

    const phone = messageData.from;
    const text = messageData.text?.body || "";
    const type = messageData.type || "text";

    console.log("📩 Incoming:", phone, text);

    const { data: member } = await supabase
      .from("members")
      .select("gym_id")
      .eq("phone", phone)
      .maybeSingle();

    await supabase.from("messages").insert([
      {
        phone,
        message: text,
        sender: "user",
        type,
        gym_id: member?.gym_id || null,
      }
    ]);

    console.log("✅ Message saved");

    res.sendStatus(200);

  } catch (err) {
    console.log("❌ Webhook Error:", err);
    res.sendStatus(500);
  }
});

// 📤 SEND MESSAGE (CHAT UI)
app.post("/send-message", async (req, res) => {
  try {
    const { phone, message } = req.body;

    if (!phone || !message) {
      return res.status(400).json({ error: "phone and message required" });
    }

    console.log("📤 Sending:", phone, message);

    // ✅ SAFE FETCH (NO CRASH)
    const { data: member } = await supabase
      .from("members")
      .select("gym_id")
      .eq("phone", phone)
      .maybeSingle();

    const gym_id = member?.gym_id || null;

    // 📡 SEND TO WHATSAPP
    await axios.post(
      `https://graph.facebook.com/v18.0/${process.env.PHONE_NUMBER_ID}/messages`,
      {
        messaging_product: "whatsapp",
        to: phone,
        type: "text",
        text: { body: message },
      },
      {
        headers: {
          Authorization: `Bearer ${process.env.ACCESS_TOKEN}`,
          "Content-Type": "application/json",
        },
      }
    );

    // 💾 SAVE TO DB (IMPORTANT)
    await supabase.from("messages").insert([
      {
        phone,
        message,
        sender: "business",
        type: "text",
        gym_id,
      },
    ]);

    console.log("✅ Message sent + saved");

    res.json({ success: true });

  } catch (err) {
    console.log("❌ ERROR:", err.response?.data || err.message);

    res.status(500).json({
      error: "Failed",
      details: err.response?.data || err.message
    });
  }
});

app.post("/send-reminder", async (req, res) => {
  try {
    const { member_id } = req.body;

    const { data: member } = await supabase
      .from("members")
      .select("*")
      .eq("id", member_id)
      .single();

    if (!member) {
      return res.status(404).json({ error: "Member not found" });
    }

    const today = new Date();
    const expiry = new Date(member.expiry_date);

    const diffDays = Math.ceil(
      (expiry - today) / (1000 * 60 * 60 * 24)
    );

    let type = "";
    let templateName = "";

    if (diffDays === 3) {
      type = "before_expiry";
      templateName = "before_expiry";
    } else if (diffDays === 0) {
      type = "today_expiry";
      templateName = "today_expiry";
    } else if (diffDays < 0) {
      type = "after_expiry";
      templateName = "after_expiry";
    } else {
      return res.json({ message: "No reminder needed" });
    }

    // 📤 SEND TEMPLATE
    await sendTemplate(
      member.phone,
      member.name,
      member.expiry_date,
      templateName
    );

    // 💾 SAVE SAME MESSAGE
    const msg = formatMessage(type, member.name, member.expiry_date);

    await supabase.from("messages").insert([
      {
        phone: member.phone,
        message: msg,
        sender: "business",
        type: "text",
        gym_id: member.gym_id,
      },
    ]);

    // 🔄 UPDATE STATUS
    await supabase
      .from("members")
      .update({ status: `${type.replace("_expiry", "")}_sent` })
      .eq("id", member.id);

    res.json({ success: true });

  } catch (err) {
    console.log("❌ Reminder Error:", err);
    res.status(500).json({ error: "Failed" });
  }
});

const PORT = 5000;
app.listen(PORT, () => {
  console.log(`🚀 Server running on port ${PORT}`);
